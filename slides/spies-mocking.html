Mocking with Spies

describe('NSA', function () {
    var safe;

    beforeEach(function () {
        safe = {
            getInfo: function () { return 'My stuff'; }
        };
        spyOn(safe, 'geInfo').andCallFake(function () {
            return 'NSA stuff';
        });
        // or shorter: spyOn(safe, 'getInfo').andReturn('NSA stuff');
    });

    it('should tell your information', function () {
        expect(safe.getInfo()).toBe('NSA stuff');
    });
});

  • .and.callFake();
  • .and.returnValue();
