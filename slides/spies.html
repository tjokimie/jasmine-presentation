Spies

describe('NSA', function () {
    var storage;

    beforeEach(function () {
        storage = {
            getInfo: function () {
                return 'Top secret stuff';
            }
        };
        spyOn(storage, 'getInfo').andCallThrough();
    });

    it('should be spying on you', function () {
        storage.getInfo();
        expect(storage.getInfo).toHaveBeenCalled();
    });
});

By chaining the spy with andCallThrough, the spy will still track all calls to it but in addition it will delegate to the actual implementation.
