Matching with Spies

describe('NSA', function () {
    var storage;

    beforeEach(function () {
        storage = {
            setInfo: function (stuff) { /* set info */ }
        };
        spyOn(storage, 'setInfo').andCallThrough();
    });                       // .and.callThrough(); in Jasmine 2.0

    it('should know your stuff', function () {
        storage.setInfo('Top secret stuff');
        expect(storage.setInfo).toHaveBeenCalledWith('Top secret stuff');
        // ...or
        expect(storage.setInfo).toHaveBeenCalledWith(jasmine.any(String));
    });
});
