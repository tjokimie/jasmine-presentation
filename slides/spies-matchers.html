Matching with Spies

describe('NSA', function () {
    var safe;

    beforeEach(function () {
        safe = {
            setInfo: function (stuff) { /* set info */ }
        };
        spyOn(safe, 'setInfo').andCallThrough();
    });                    // .and.callThrough(); in Jasmine 2.0

    it('should know your stuff', function () {
        safe.setInfo('Top secret stuff');
        expect(safe.setInfo).toHaveBeenCalledWith('Top secret stuff');
        // ...or
        expect(safe.setInfo).toHaveBeenCalledWith(jasmine.any(String));
    });
});
